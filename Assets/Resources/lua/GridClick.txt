import "UnityEngine"

if not UnityEngine.GameObject or not  UnityEngine.UI then
	error("Click Make/All to generate lua wrap file")
end

local class={}

function OnClick()
	class.gridClick.isClick = not(class.gridClick.isClick)
	if class.gridClick.isClick then
		class.image.color=Color.black
	else
		class.image.color=Color.white
	end
end

function class:UpdateClick()
	class.gridClick.isClick = not(class.gridClick.isClick)
	if class.gridClick.isClick then
		class.image.color=Color.black
	else
		class.image.color=Color.white
	end
end


function main()
	return class
end

function class:init(transform)
	if transform == nil then
		return print("transform is nil")
	end
	class.gridClick=transform:GetComponent("GridClick")
	class.image=transform:GetComponent("Image")
	class.button=transform:GetComponent(UI.Button)
	class.button.onClick:AddListener(OnClick)
end
